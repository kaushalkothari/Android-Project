Algorith For Insertion

void insert(node *ptr, int data)
{
/* Iterate through the list till we encounter the last node.*/
while(ptr->next!=NULL)
{
ptr = ptr -> next;
}
/* Allocate memory for the new node and put data in it.*/
ptr->next = (node *)malloc(sizeof(node));
ptr = ptr->next;
ptr->data = data;
ptr->next = NULL;
}

Algorith For Delete Begining

void deleteatbeg(node **start)
{

}

Algorith For Delete Last

void deleteatlast(node **start)
{
node *ptr,*cptr;
int temp;
if((*start)->next == NULL)
{
ptr = *start ;
*start = NULL;
temp = ptr->info;
}
else
{
cptr = *start ;
ptr =(*start)->next;
while(ptr->next != NULL)
{
cptr = ptr;
ptr = ptr->next;
}
cptr->next = NULL;
temp = ptr->info;
}
free(ptr);
printf("\nDeleted item is %d : \n",temp);
}
